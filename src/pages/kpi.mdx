## Post Generate Denda . {{ tag: 'POST', label: '/v1/human-resource/generateDenda' }}
<Row>
    <Col>

    Endpoint for generate denda.


     <Properties>
           This endpoint for generate denda.
    </Properties>

    # Request

    ### Required attributes

     <Properties>
        <Property name="employee_id" type="string">
           Unique identifier for the employee.
        </Property>
        <Property name="submission_date" type="string (YYYY-MM-DD)">
            The submission_date The date when the employee submitted their evaluation.
        </Property>
        <Property name="deadline_date" type="string (YYYY-MM-DD)">
            The deadline_date The deadline for submitting evaluations.
        </Property>
        <Property name="status" type="string (draft, submitted)">
            The statys drafy and submitted The status of the evaluation (draft or submitted).
        </Property>
        <Property name="penalty_basis" type="number">
            The penalty_basis The monetary basis for calculating the penalty per day..
        </Property>
    </Properties>

    ### Response

    <Properties>
        <Property name="employee_id" type="string">
           The ID of the employee.
        </Property>
        <Property name="penalty_amount" type="number">
            Total penalty amount in Rupiah.
        </Property>
        <Property name="days_late" type="number">
            Total number of days late.
        </Property>
        <Property name="evaluation_status" type="string">
            Status of the evaluation (submitted, draft, not submitted)..
        </Property>
        <Property name="message" type="string">
            Additional information about the penalty calculation.
        </Property>
    </Properties>

    ### Notes
    <Properties>
        <Property name="note">
           If the penalty basis is not provided or is null, the penalty amount will also be considered null.
        </Property>
        <Property name="note">

        </Property>
    </Properties>

## History Denda. {{ tag: 'GET', label: '/v1/human-resource/historyDenda?employee_id=E123&start_date=2023-08-01&end_date=2023-09-05' }}
<Row>
    <Col>

    Endpoint for histoy denda.


     <Properties>
           This endpoint for histoy denda.
    </Properties>

    ### Response

    <Properties>
        <Property name="Description">
           Retrieves a history of late evaluation penalties for employees, with default filters applied.
        </Property>
        <Property name="Query Parameters">
           employee_id (string, optional): Filter by specific employee ID.
        </Property>
        <Property name="Query Parameters">
            status (string, optional): Filter by evaluation status (e.g., draft, submitted, not submitted).
        </Property>
        <Property name="Query Parameters">
            start_date (string, optional): Filter report from this date (YYYY-MM-DD).
        </Property>
        <Property name="Query Parameters">
            end_date (string, optional): Filter report until this date (YYYY-MM-DD).
        </Property>
        <Property name="Default Filters">
            If no filters are provided, the API will default to returning penalties for the last 30 days.
        </Property>
        <Property name="Default Filters">
            The default status filter will be set to include all statuses.
        </Property>
    </Properties>

    ### Notes
    <Properties>
        <Property name="note">
           If no penalties are found for the specified filters, the response should contain an empty history array and an appropriate message.
        </Property>
        <Property name="note">
            The API should handle date validation and return errors for invalid date formats.
        </Property>
        <Property name="note">
            Pagination may be implemented if the history data set is extensive, to improve performance.
        </Property>
    </Properties>
